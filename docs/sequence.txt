' == Sequence: Checkout & Membuat Pesanan ==
@startuml sequence_uc4_checkout
actor Pelanggan
participant "Halaman Checkout" as UI
participant "OrdersService\n(addOrderAndTransaction)" as Service
database "Firestore\norders + transactions" as DB

Pelanggan -> UI : Isi form & klik Lanjutkan
activate UI
UI -> UI : validate(form)
alt valid
  UI -> UI : generateOrderCode()
  UI -> Service : addOrderAndTransaction(code, order, txn)
  activate Service
  Service -> DB : writeBatch.set(orders/code) + set(transactions)
  activate DB
  DB --> Service : commit OK
  deactivate DB
  Service --> UI : orderCode
  deactivate Service
  UI -> UI : clear cart
  UI --> Pelanggan : Redirect ke /checkout/success?code=...
else tidak valid
  UI --> Pelanggan : Tampilkan error field
end
deactivate UI
@enduml
